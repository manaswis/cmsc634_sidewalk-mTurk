# this terminal command will query the sidewalk database and output the label_data.csv file in your current dir, copy into cmsc634_sidewalk-mTurk/data

psql -d sidewalk -t -A -F"," -c "SELECT sidewalk.label_point.lng, sidewalk.label_point.lat, sidewalk.label_type.label_type, sidewalk.audit_task.user_id FROM sidewalk.label INNER JOIN sidewalk.audit_task  ON sidewalk.label.audit_task_id = sidewalk.audit_task.audit_task_id INNER JOIN sidewalk.label_point  ON sidewalk.label.label_id = sidewalk.label_point.label_id INNER JOIN sidewalk.label_type  ON sidewalk.label.label_type_id = sidewalk.label_type.label_type_id WHERE (sidewalk.audit_task.completed = true) AND (sidewalk.label.deleted = false)" > labels-test.csv
